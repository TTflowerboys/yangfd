Life Cycle Events

  1. deviceready

    Cordova native code is initialized and Cordova APIs can be called from JavaScript

  2. viewreappear

    View Controller appear again, the first time cordova may not load finished, so first time appear, not call this

  3. viewdisappear

    View Controller disppear

  4. window.onunload

    dom onunload, can do saving stateâ€¨

Example Code

<script type="text/javascript" charset="utf-8">
window.onunload = function() {

}
(function () {
 document.addEventListener('deviceready', onDeviceReady, false);

function onDeviceReady() {
    window.alert('view deviceready')
}


document.addEventListener('viewreappear', onViewReappear, false);

function onViewReappear() {

    window.alert('view reappear')
}

document.addEventListener('viewdisappear', onViewDisappear, false);

function onViewDisappear() {

    window.alert('view disappear')
}
})()
</script>
